#industrial bill

country_event = { 
	id = poland.2
	title = poland.2.t
	desc = poland.2.d
	picture = GFX_report_event_generic_factory

	mean_time_to_happen = { days = 1 }

	is_triggered_only = yes

	option = {
		name = poland.2.a
		ai_chance = {
			factor = 40
		}
		add_popularity = {
			ideology = authoritarian_democrat
			popularity = 0.02
		}
		add_ideas = POL_MOW_industrial
		random_owned_controlled_state = {
			limit = {
				free_building_slots = {
				building = arms_factory
				size > 0
				include_locked = yes
				}
			}
			add_extra_state_shared_building_slots = 3
			add_building_construction = {
			type = industrial_complex
			level = 3
			instant_build = yes
		}
	}
}
	option = {
		name = poland.2.b
		ai_chance = {
			factor = 40
		}
		add_popularity = {
			ideology = social_conservative
			popularity = 0.02
		}
		add_ideas = POL_PSCD_industrial
		

	}
	option = {
		name = poland.2.c
		
		ai_chance = {
			factor = 20
			}
		add_political_power = -40

		hidden_effect = {
		if = {
			limit = {
				POL = { NOT = { has_country_flag = veto1}}
			}
			set_country_flag = veto1
			}
		if = {
			limit = {
				POL = has_country_flag = veto1
			}
			set_country_flag = veto2
			}
		if = {
			limit = {
				POL = has_country_flag = veto2
			}
			set_country_flag = veto3
			}
		if = {
			limit = {
				POL = has_country_flag = veto3
			}
			set_country_flag = veto4
			}
		if = {
			limit = {
				POL = has_country_flag = veto4
			}
			set_country_flag = veto5
			}
		if = {
			limit = {
				POL = has_country_flag = veto5
			}
			set_country_flag = veto6
			}
		if = {
			limit = {
				POL = has_country_flag = veto6
			}

			}
		if = {
