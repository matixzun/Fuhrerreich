# Serbia Demands Montenegro
country_event = {
	id = serbia.6
	title = serbia.6.t
	desc = serbia.6.d
	picture = GFX_report_event_hitler_parade
	
	is_triggered_only = yes

	fire_only_once = yes

	  trigger = {
	    has_country_flag = serbia_demands_montenegro
	  }
	}
	
	option = { # Accept
		name = serbia.6.a
		ai_chance = {
			base = 30
			modifier = {
				add = -25
				SER = { has_army_size = { size < 5 } }
			}
			modifier = {
				add = 100
				SER = { has_army_size = { size > 10 } }
			}
			modifier = {
				add = 20
				is_in_faction_with = SER
			}
		}
		SER = {
			country_event = { hours = 2 id = serbia.5 }
		}
		custom_effect_tooltip = GAME_OVER_TT
	}
	option = { # Refuse
		name = serbia.6.b
		ai_chance = {
			base = 10			
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
			modifier = {
				SER = { has_army_size = { size > 10 } }
				factor = 0.5
			}
			modifier = {
				add = 20
				SER = { has_army_size = { size < 5 } }
			}
			modifier = {
				factor = 2
				is_in_faction = yes
				NOT = { is_in_faction_with = SER }
			}
			modifier = {
				any_other_country = {
					has_guaranteed = SER
				}
				factor = 5
			}
		}
		SER = {
			add_opinion_modifier = { target = ROOT modifier = SER_montenegro_annexation_rejected }
			country_event = { hours = 2 id = serbia.7 }
			if = { limit = { is_in_faction_with = MNT }
				remove_from_faction = MNT
			}
		}
	}
}
